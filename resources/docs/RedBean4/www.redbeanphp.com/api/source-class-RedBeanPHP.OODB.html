132:     /**
133:      * Toggles fluid or frozen mode. In fluid mode the database
134:      * structure is adjusted to accomodate your objects. In frozen mode
135:      * this is not the case.
136:      *
137:      * You can also pass an array containing a selection of frozen types.
138:      * Let's call this chilly mode, it's just like fluid mode except that
139:      * certain types (i.e. tables) aren't touched.
140:      *
141:      * @param boolean|array $toggle TRUE if you want to use OODB instance in frozen mode
142:      *
143:      * @return void
144:      */
145:     public function freeze( $toggle )
146:     {
147:         if ( is_array( $toggle ) ) {
148:             $this->chillList = $toggle;
149:             $this->isFrozen  = FALSE;
150:         } else {
151:             $this->isFrozen = (boolean) $toggle;
152:         }
153: 
154:         if ( $this->isFrozen ) {
155:             if ( !$this->frozenRepository ) {
156:                 $this->frozenRepository = new FrozenRepo( $this, $this->writer );
157:             }
158: 
159:             $this->repository = $this->frozenRepository;
160: 
161:         } else {
162:             if ( !$this->fluidRepository ) {
163:                 $this->fluidRepository = new FluidRepo( $this, $this->writer );
164:             }
165: 
166:             $this->repository = $this->fluidRepository;
167:         }
168: 
169:         if ( count( self::$sqlFilters ) ) {
170:             AQueryWriter::setSQLFilters( self::$sqlFilters, ( !$this->isFrozen ) );
171:         }
172:

517:     /**
518:      * Binds an SQL function to a column.
519:      * This method can be used to setup a decode/encode scheme or
520:      * perform UUID insertion. This method is especially useful for handling
521:      * MySQL spatial columns, because they need to be processed first using
522:      * the asText/GeomFromText functions.
523:      *
524:      * @param string $mode (read or write)
525:      * @param string $field
526:      * @param string $function
527:      */
528:     public function bindFunc( $mode, $field, $function )
529:     {
530:         list( $type, $property ) = explode( '.', $field );
531:         $mode = ($mode === 'write') ? QueryWriter::C_SQLFILTER_WRITE : QueryWriter::C_SQLFILTER_READ;
532: 
533:         if ( !isset( self::$sqlFilters[$mode] ) ) self::$sqlFilters[$mode] = array();
534:         if ( !isset( self::$sqlFilters[$mode][$type] ) ) self::$sqlFilters[$mode][$type] = array();
535: 
536:         if ( is_null( $function ) ) {
537:             unset( self::$sqlFilters[$mode][$type][$property] );
538:         } else {
539:             if ($mode === QueryWriter::C_SQLFILTER_WRITE) {
540:                 self::$sqlFilters[$mode][$type][$property] = $function.'(?)';
541:             } else {
542:                 self::$sqlFilters[$mode][$type][$property] = $function."($field)";
543:             }
544:         }
545: 
546:         AQueryWriter::setSQLFilters( self::$sqlFilters, ( !$this->isFrozen ) );
547:     }

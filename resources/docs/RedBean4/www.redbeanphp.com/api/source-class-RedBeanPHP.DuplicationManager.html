 62:      * Copies the shared beans in a bean, i.e. all the sharedBean-lists.
 63:      *
 64:      * @param OODBBean $copy   target bean to copy lists to

 79:     /**
 80:      * Copies the own beans in a bean, i.e. all the ownBean-lists.
 81:      * Each bean in the own-list belongs exclusively to its owner so
 82:      * we need to invoke the duplicate method again to duplicate each bean here.
 83:      *
 84:      * @param OODBBean $copy        target bean to copy lists to
 85:      * @param string   $owned       name of the own list
 86:      * @param array    $beans       array with shared beans to copy
 87:      * @param array    $trail       array with former beans to detect recursion
 88:      * @param boolean  $preserveIDs TRUE means preserve IDs, for export only
 89:      *
 90:      * @return void
 91:      */
 92:     private function copyOwnBeans( OODBBean $copy, $owned, $beans, $trail, $preserveIDs )
 93:     {
 94:         $copy->$owned = array();
 95:         foreach ( $beans as $subBean ) {
 96:             array_push( $copy->$owned, $this->duplicate( $subBean, $trail, $preserveIDs ) );
 97:         }

122:     /**
123:      * Generates a key from the bean type and its ID and determines if the bean

192:     /**
193:      * @see DuplicationManager::dup
194:      *
195:      * @param OODBBean $bean        bean to be copied
196:      * @param array    $trail       trail to prevent infinite loops
197:      * @param boolean  $preserveIDs preserve IDs
198:      *
199:      * @return OODBBean
200:      */
201:     protected function duplicate( OODBBean $bean, $trail = array(), $preserveIDs = FALSE )
202:     {
203:         if ( $this->inTrailOrAdd( $trail, $bean ) ) return $bean;
204: 
205:         $type = $bean->getMeta( 'type' );
206: 
207:         $copy = $this->createCopy( $bean );
208:         foreach ( $this->tables as $table ) {
209: 
210:             if ( !empty( $this->filters ) ) {
211:                 if ( !in_array( $table, $this->filters ) ) continue;
212:             }
213: 
214:             list( $owned, $shared ) = $this->getListNames( $table );
215: 
216:             if ( $this->hasSharedList( $type, $table ) ) {
217:                 if ( $beans = $bean->$shared ) {
218:                     $this->copySharedBeans( $copy, $shared, $beans );
219:                 }
220:             } elseif ( $this->hasOwnList( $type, $table ) ) {
221:                 if ( $beans = $bean->$owned ) {
222:                     $this->copyOwnBeans( $copy, $owned, $beans, $trail, $preserveIDs );
223:                 }
224: 
225:                 $copy->setMeta( 'sys.shadow.' . $owned, NULL );
226:             }
227: 
228:             $copy->setMeta( 'sys.shadow.' . $shared, NULL );
229:         }
230: 
231:         $copy->id = ( $preserveIDs ) ? $bean->id : $copy->id;
232: 
233:         return $copy;
234:     }

250:      * Recursively turns the keys of an array into
251:      * camelCase.
252:      *
253:      * @param array   $array       array to camelize
254:      * @param boolean $dolphinMode whether you want the exception for IDs.
255:      *
256:      * @return array
257:      */
258:     public function camelfy( $array, $dolphinMode = false ) {
259:         $newArray = array();
260:         foreach( $array as $key => $element ) {
261:             $newKey = preg_replace_callback( '/_(\w)/', function( &$matches ){
262:                 return strtoupper( $matches[1] );
263:             }, $key);
264: 
265:             if ( $dolphinMode ) {
266:                 $newKey = preg_replace( '/(\w)Id$/', '$1ID', $newKey );
267:             }
268: 
269:             $newArray[$newKey] = ( is_array($element) ) ? $this->camelfy( $element, $dolphinMode ) : $element;
270:         }
271:         return $newArray;
272:     }

348:     /**
349:      * Makes a copy of a bean. This method makes a deep copy
350:      * of the bean.The copy will have the following features.
351:      * - All beans in own-lists will be duplicated as well
352:      * - All references to shared beans will be copied but not the shared beans themselves
353:      * - All references to parent objects (_id fields) will be copied but not the parents themselves
354:      * In most cases this is the desired scenario for copying beans.
355:      * This function uses a trail-array to prevent infinite recursion, if a recursive bean is found
356:      * (i.e. one that already has been processed) the ID of the bean will be returned.
357:      * This should not happen though.
358:      *
359:      * Note:
360:      * This function does a reflectional database query so it may be slow.
361:      *
362:      * Note:
363:      * this function actually passes the arguments to a protected function called
364:      * duplicate() that does all the work. This method takes care of creating a clone
365:      * of the bean to avoid the bean getting tainted (triggering saving when storing it).
366:      *
367:      * @param OODBBean $bean        bean to be copied
368:      * @param array    $trail       for internal usage, pass array()
369:      * @param boolean  $preserveIDs for internal usage
370:      *
371:      * @return OODBBean
372:      */
373:     public function dup( OODBBean $bean, $trail = array(), $preserveIDs = FALSE )
374:     {
375:         if ( !count( $this->tables ) ) {
376:             $this->tables = $this->toolbox->getWriter()->getTables();
377:         }
378: 
379:         if ( !count( $this->columns ) ) {
380:             foreach ( $this->tables as $table ) {
381:                 $this->columns[$table] = $this->toolbox->getWriter()->getColumns( $table );
382:             }
383:         }
384: 
385:         $rs = $this->duplicate( ( clone $bean ), $trail, $preserveIDs );
386: 
387:         if ( !$this->cacheTables ) {
388:             $this->tables  = array();
389:             $this->columns = array();
390:         }
391: 
392:         return $rs;
393:     }

395:     /**
396:      * Exports a collection of beans recursively.
397:      * This method will export an array of beans in the first argument to a
398:      * set of arrays. This can be used to send JSON or XML representations
399:      * of bean hierarchies to the client.
400:      *
401:      * For every bean in the array this method will export:
402:      *
403:      * - contents of the bean
404:      * - all own bean lists (recursively)
405:      * - all shared beans (but not THEIR own lists)
406:      *
407:      * If the second parameter is set to TRUE the parents of the beans in the
408:      * array will be exported as well (but not THEIR parents).
409:      *
410:      * The third parameter can be used to provide a white-list array
411:      * for filtering. This is an array of strings representing type names,
412:      * only the type names in the filter list will be exported.
413:      *
414:      * The fourth parameter can be used to change the keys of the resulting
415:      * export arrays. The default mode is 'snake case' but this leaves the
416:      * keys as-is, because 'snake' is the default case style used by
417:      * RedBeanPHP in the database. You can set this to 'camel' for
418:      * camel cased keys or 'dolphin' (same as camelcase but id will be
419:      * converted to ID instead of Id).
420:      *
421:      * @param array|OODBBean $beans     beans to be exported
422:      * @param boolean        $parents   also export parents
423:      * @param array          $filters   only these types (whitelist)
424:      * @param string         $caseStyle case style identifier
425:      *
426:      * @return array
427:      */
428:     public function exportAll( $beans, $parents = FALSE, $filters = array(), $caseStyle = 'snake')
429:     {
430:         $array = array();
431: 
432:         if ( !is_array( $beans ) ) {
433:             $beans = array( $beans );
434:         }
435: 
436:         foreach ( $beans as $bean ) {
437:             $this->setFilters( $filters );
438: 
439:             $duplicate = $this->dup( $bean, array(), TRUE );
440: 
441:             $array[]   = $duplicate->export( FALSE, $parents, FALSE, $filters );
442:         }
443: 
444:         if ( $caseStyle === 'camel' ) $array = $this->camelfy( $array );
445:         if ( $caseStyle === 'dolphin' ) $array = $this->camelfy( $array, true );
446: 
447:         return $array;
448:     }
